// 清除页面上所有已有的内容
const container = this.container;
container.innerHTML = '';

// 获取搜索词并去除首尾空格
let searchTerm = (dv.current()?.img || "").trim();

// 解析搜索语法
function parseSearchTerms(query) {
    if (!query) return null;
    
    // 分解搜索词
    let terms = {
        include: [], // AND 条件
        or: [],      // OR (|) 条件
        exclude: []  // NOT 条件
    };
    
    // 分割 | 条件
    let orGroups = query.split('|');
    
    orGroups.forEach(group => {
        // 处理每个 | 组内的条件
        let words = group.trim().split(' ');
        
        words.forEach(word => {
            word = word.trim().toLowerCase();
            if (!word) return;
            
            if (word.startsWith('-')) {
                // NOT 条件
                let excludeTerm = word.slice(1);
                if (excludeTerm) {
                    terms.exclude.push(excludeTerm);
                }
            } else if (group === orGroups[0]) {
                // 第一个 | 组中的非排除词作为 AND 条件
                terms.include.push(word);
            } else {
                // 其他 | 组中的词作为 OR 条件
                terms.or.push(word);
            }
        });
    });
    
    return terms;
}

// 检查文件名是否匹配搜索条件
function matchesSearchTerms(fileName, terms) {
    if (!terms) return false;
    
    fileName = fileName.toLowerCase();
    
    // 检查排除条件
    for (let exclude of terms.exclude) {
        if (fileName.includes(exclude)) return false;
    }
    
    // 检查必须包含的条件
    let includeMatch = terms.include.length === 0 || 
        terms.include.every(term => fileName.includes(term));
    
    // 检查 OR 条件
    let orMatch = terms.or.length === 0 || 
        terms.or.some(term => fileName.includes(term));
    
    return includeMatch && orMatch;
}

// 定义支持的图片类型
const imageTypes = ['.jpg', '.jpeg', '.png', '.gif', '.webp', '.bmp', '.svg'];

// 创建自定义样式
const style = document.createElement('style');

// 创建搜索帮助信息
if (!searchTerm) {
    const helpText = document.createElement('div');
    helpText.className = 'search-help';
    helpText.innerHTML = `
        <div>支持的搜索语法：</div>
        <ul style="margin: 5px 0 0 20px; padding: 0;">
            <li>AND：空格分隔关键词（例如：cat dog）</li>
            <li>OR：使用竖线 | （例如：cat|dog）</li>
            <li>NOT：使用减号 -（例如：cat -black）</li>
        </ul>
        <div style="margin-top: 5px;">示例：cat dog|white -black</div>
        <div style="margin-top: 5px; font-style: italic;">将搜索：(包含"cat"和"dog") 或 包含"white" 但都不包含"black"的图片</div>
    `;
    container.appendChild(helpText);

// 创建图片网格容器
const gridContainer = document.createElement('div');
gridContainer.className = 'image-grid-container';
container.appendChild(gridContainer);

// 解析搜索条件
const searchTerms = parseSearchTerms(searchTerm);

if (!searchTerm) {
    // 显示提示文字
    const placeholder = document.createElement('div');
    placeholder.className = 'placeholder-text';
    placeholder.textContent = '请输入搜索词以查找图片';
    container.appendChild(placeholder);
} else {
    // 获取并过滤文件
    let matchedFiles = app.vault.getAllLoadedFiles("").filter(file => 
        imageTypes.some(type => file.path.toLowerCase().endsWith(type)) &&
        matchesSearchTerms(file.name, searchTerms)
    );

    if (matchedFiles.length === 0) {
        const noResults = document.createElement('div');
        noResults.className = 'placeholder-text';
        noResults.textContent = '未找到匹配的图片文件';
        container.appendChild(noResults);
    } else {
        // 渲染找到的图片
        matchedFiles.forEach(file => {
            const card = document.createElement('div');
            card.className = 'image-card';

            // 复制按钮
            const copyButton = document.createElement('button');
            copyButton.className = 'copy-button';
            copyButton.textContent = '复制链接';
            copyButton.onclick = (e) => {
                e.stopPropagation();
                navigator.clipboard.writeText(`![[${file.name}]]`).then(() => {
                    const toast = document.createElement('div');
                    toast.style.cssText = `
                        position: fixed;
                        bottom: 20px;
                        left: 50%;
                        transform: translateX(-50%);
                        background: rgba(0,0,0,0.7);
                        color: white;
                        padding: 8px 16px;
                        border-radius: 4px;
                        z-index: 1000;
                    `;
                    toast.textContent = '已复制到剪贴板';
                    document.body.appendChild(toast);
                    setTimeout(() => toast.remove(), 2000);
                });
            };
            card.appendChild(copyButton);

            // 图片容器
            const imgContainer = document.createElement('div');
            imgContainer.className = 'image-container';
            
            // 图片
            const img = document.createElement('img');
            img.className = 'image-preview';
            img.src = app.vault.adapter.getResourcePath(file.path);
            img.alt = file.name;
            imgContainer.appendChild(img);
            card.appendChild(imgContainer);

            // 文件名
            const fileName = document.createElement('div');
            fileName.className = 'file-name';
            const extension = file.path.split('.').pop().toLowerCase();
            fileName.innerHTML = `[[${file.name}]] (${extension})`;
            card.appendChild(fileName);

            gridContainer.appendChild(card);
        });

        // 底部提示
        const hint = document.createElement('div');
        hint.className = 'placeholder-text';
        container.appendChild(hint);
    }
}
