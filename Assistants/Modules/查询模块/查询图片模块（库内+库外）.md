// Path handling utility
class PathUtils {
  static parseInlineSelect(input) {
    if (!input?.trim()) return [];
    const optionRegex = /option\((.*?)\)/g;
    try {
      return Array.from(input.matchAll(optionRegex), match => match[1]);
    } catch (error) {
      console.error('Error parsing inlineSelect:', error);
      return [];
    }
  }

  static getSelectedPath(input) {
    if (!input?.trim()) return '';
    if (input.includes('INPUT[inlineSelect')) {
      const options = this.parseInlineSelect(input);
      return options[options.length - 1] || '';
    }
    return input;
  }

  static isExternalPath(path) {
    if (!path?.trim()) return false;
    const actualPath = this.getSelectedPath(path);
    return /^[A-Za-z]:\\/.test(actualPath) || actualPath.startsWith('/');
  }

  static async toImageUrl(filePath, isExternal = false) {
    if (!filePath?.trim()) return '';
    
    try {
      if (!isExternal) {
        return app.vault.adapter.getResourcePath(filePath);
      }
      
      // 处理外部文件路径
      const fs = require('fs').promises;
      const imageBuffer = await fs.readFile(filePath);
      const base64Image = imageBuffer.toString('base64');
      const extension = filePath.split('.').pop().toLowerCase();
      return `data:image/${extension};base64,${base64Image}`;
    } catch (error) {
      console.error('Error converting path to URL:', error);
      return '';
    }
  }

  static getMarkdownLink(file) {
    if (file.isExternal) {
      return `![${file.name}](${file.path})`;
    }
    return `![[${file.name}]]`;
  }

  static isImageFile(filename) {
    return CONFIG.imageTypes.some(ext => 
      filename.toLowerCase().endsWith(ext)
    );
  }
}

// File system operations for external files
class FileSystemAccess {
  static async listDirectory(path) {
    try {
      const fs = require('fs').promises;
      const pathModule = require('path');
      
      const stats = await fs.stat(path);
      
      if (stats.isFile()) {
        return PathUtils.isImageFile(path) 
          ? [{
              name: pathModule.basename(path),
              path: path,
              isExternal: true
            }]
          : [];
      }
      
      const entries = await fs.readdir(path, { withFileTypes: true });
      return (await Promise.all(entries.map(async entry => {
        const fullPath = pathModule.join(path, entry.name);
        if (entry.isFile() && PathUtils.isImageFile(entry.name)) {
          return [{
            name: entry.name,
            path: fullPath,
            isExternal: true
          }];
        }
        if (entry.isDirectory()) {
          return await this.listDirectory(fullPath);
        }
        return [];
      }))).flat();
      
    } catch (error) {
      console.error('Error reading directory:', error);
      return [];
    }
  }
}

// Search functionality
class SearchEngine {
  #terms = null;

  constructor(query) {
    this.#terms = this.#parseSearchTerms(query);
  }

  #parseSearchTerms(query) {
    if (!query?.trim()) return null;
    
    const terms = {
      include: [],
      or: [],
      exclude: []
    };
    
    query.split('|').forEach((group, index) => {
      group.trim().split(' ').forEach(word => {
        word = word.trim().toLowerCase();
        if (!word) return;
        
        if (word.startsWith('-')) {
          const excludeTerm = word.slice(1);
          if (excludeTerm) terms.exclude.push(excludeTerm);
        } else if (index === 0) {
          terms.include.push(word);
        } else {
          terms.or.push(word);
        }
      });
    });
    
    return terms;
  }

  matchesSearchTerms(fileName) {
    if (!this.#terms) return true;
    
    fileName = fileName.toLowerCase();
    
    if (this.#terms.exclude.some(term => fileName.includes(term))) {
      return false;
    }
    
    const includeMatch = this.#terms.include.length === 0 || 
      this.#terms.include.every(term => fileName.includes(term));
    
    const orMatch = this.#terms.or.length === 0 || 
      this.#terms.or.some(term => fileName.includes(term));
    
    return includeMatch && orMatch;
  }
}

// Main Image Browser Class
class ImageBrowser {
  #container;
  #gridContainer;
  #searchEngine;
  #searchPath;
  #searchTerm;

  constructor(container, currentPage) {
    this.#container = container;
    this.#searchTerm = (currentPage.img || '').trim();
    this.#searchPath = (currentPage.selectPath || '').trim();
    this.#searchEngine = new SearchEngine(this.#searchTerm);
    
    this.#initializeUI();
    this.#initializeSearch();
  }

  #initializeUI() {
    this.#container.innerHTML = '';
    document.head.appendChild(UIComponents.createStyleSheet());
    
    this.#container.appendChild(
      UIComponents.createSearchInfo(this.#searchPath, this.#searchTerm)
    );
    
    this.#gridContainer = document.createElement('div');
    this.#gridContainer.className = 'imageGrid';
    this.#container.appendChild(this.#gridContainer);
  }

  async #searchImages() {
    try {
      // 如果没有搜索词，清空图片显示并返回
      if (!this.#searchTerm) {
        this.#gridContainer.innerHTML = '';
        const placeholder = document.createElement('div');
        placeholder.className = 'placeholderText';
        placeholder.textContent = '请输入搜索关键词';
        this.#gridContainer.appendChild(placeholder);
        return;
      }

      if (!this.#searchPath) {
        throw new Error('未选择搜索路径');
      }

      const actualPath = PathUtils.getSelectedPath(this.#searchPath);
      let matchedFiles = [];

      if (PathUtils.isExternalPath(actualPath)) {
        // 处理外部文件路径
        matchedFiles = await FileSystemAccess.listDirectory(actualPath);
      } else {
        // 处理库内文件
        matchedFiles = app.vault.getFiles()
          .filter(file => 
            file.path?.startsWith(actualPath) &&
            PathUtils.isImageFile(file.path)
          )
          .map(file => ({
            name: file.name,
            path: file.path,
            isExternal: false
          }));
      }

      matchedFiles = matchedFiles.filter(file => 
        this.#searchEngine.matchesSearchTerms(file.name)
      );

      await this.#renderResults(matchedFiles);
    } catch (error) {
      console.error('Search error:', error);
      this.#showError(error.message);
    }
  }

  async #renderResults(matchedFiles) {
    this.#gridContainer.innerHTML = '';
    
    if (matchedFiles.length === 0) {
      this.#gridContainer.appendChild(UIComponents.createNoResultsPlaceholder());
      return;
    }

    for (const file of matchedFiles) {
      await this.#createImageCard(file);
    }
  }

  async #createImageCard(file) {
    const card = document.createElement('div');
    card.className = 'imageCard';

    const copyButton = document.createElement('button');
    copyButton.className = 'copyButton';
    copyButton.textContent = '复制链接';
    copyButton.onclick = (e) => {
      e.stopPropagation();
      navigator.clipboard.writeText(PathUtils.getMarkdownLink(file))
        .then(() => UIComponents.createToast('已复制到剪贴板'));
    };
    card.appendChild(copyButton);

    const imgContainer = document.createElement('div');
    imgContainer.className = 'imageContainer';
    
    const img = document.createElement('img');
    img.className = 'imagePreview';
    img.style.opacity = '0';
    img.style.transition = 'opacity 0.3s';
    
    try {
      img.src = await PathUtils.toImageUrl(file.path, file.isExternal);
    } catch (error) {
      console.error('Failed to load image:', file.path);
      img.src = UIComponents.createImagePlaceholder();
    }
    
    img.alt = file.name;
    img.onload = () => img.style.opacity = '1';
    img.onerror = () => {
      console.error('Failed to load image:', file.path);
      img.src = UIComponents.createImagePlaceholder();
      img.style.opacity = '1';
    };
    
    imgContainer.appendChild(img);
    card.appendChild(imgContainer);

    const fileName = document.createElement('div');
    fileName.className = 'fileName';
    fileName.textContent = file.name;
    card.appendChild(fileName);

    this.#gridContainer.appendChild(card);
  }

  #showError(message) {
    this.#gridContainer.innerHTML = '';
    this.#gridContainer.appendChild(UIComponents.createErrorPlaceholder(message));
  }

  #initializeSearch() {
    if (this.#searchPath || this.#searchTerm) {
      this.#searchImages();
    }

    setInterval(() => {
      const newPath = (dv.current().selectPath || '').trim();
      if (newPath !== this.#searchPath) {
        this.#searchPath = newPath;
        if (this.#searchPath) {
          this.#searchImages();
        }
      }
    }, CONFIG.refreshInterval);
  }
}

this.container.innerHTML = '';
