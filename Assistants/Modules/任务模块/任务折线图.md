```dataviewjs
const allTasks = dv.pages().file.tasks;
let dateStats = [];
const today = new Date();
today.setHours(0, 0, 0, 0);

for (let i = 29; i >= 0; i--) {
    const date = new Date(today);
    date.setDate(date.getDate() - i);
    dateStats.push({
        date: date.toISOString().split('T')[0],
        total: 0,
        completed: 0
    });
}

console.log("Total tasks:", allTasks.length);

allTasks.forEach(task => {
    const taskDate = task.completion ? new Date(task.completion) : (task.due ? new Date(task.due) : null);
    if (!taskDate) return;
    
    taskDate.setHours(0, 0, 0, 0);
    const dateStr = taskDate.toISOString().split('T')[0];
    
    const statDay = dateStats.find(stat => stat.date === dateStr);
    if (statDay) {
        statDay.total++;
        if (task.completed) {
            statDay.completed++;
        }
    }
});

console.log("Date stats:", dateStats);

const baseUrl = 'https://ytmaps.vip/task-trend.html';
const stats = encodeURIComponent(JSON.stringify(dateStats));
const widgetUrl = `${baseUrl}?stats=${stats}`;

dv.paragraph(`<div style="width:100%;height:230px;overflow:hidden;">
    <iframe src="${widgetUrl}" style="width:100%;height:100%;border:none;">
    </iframe>
</div>`);
```
