```dataviewjs
function isDateInRange(date, startDays, endDays) {
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    const start = new Date(today);
    const end = new Date(today);
    start.setDate(start.getDate() + startDays);
    end.setDate(end.getDate() + endDays);
    return date >= start && date <= end;
}

const allTasks = dv.pages().file.tasks;

let taskStats = {
    today: { total: 0, completed: 0 },
    upcoming3days: { total: 0, completed: 0 },
    upcoming7days: { total: 0, completed: 0 },
    overdue: { total: 0, completed: 0 },
    noDate: { total: 0, completed: 0 }
};

allTasks.forEach(task => {
    if (!task.due) {
        taskStats.noDate.total++;
        if (task.completed) taskStats.noDate.completed++;
        return;
    }

    const dueDate = new Date(task.due);
    dueDate.setHours(0, 0, 0, 0);
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    if (dueDate < today) {
        taskStats.overdue.total++;
        if (task.completed) taskStats.overdue.completed++;
    } else if (dueDate.getTime() === today.getTime()) {
        taskStats.today.total++;
        if (task.completed) taskStats.today.completed++;
    } else if (isDateInRange(dueDate, 1, 3)) {
        taskStats.upcoming3days.total++;
        if (task.completed) taskStats.upcoming3days.completed++;
    } else if (isDateInRange(dueDate, 4, 7)) {
        taskStats.upcoming7days.total++;
        if (task.completed) taskStats.upcoming7days.completed++;
    }
});

const baseUrl = 'https://ytmaps.vip/task-dashboard.html';
const stats = encodeURIComponent(JSON.stringify(taskStats));
const widgetUrl = `${baseUrl}?stats=${stats}`;

dv.paragraph(`<div style="width:100%;height:160px;overflow:hidden;"> <iframe src="${widgetUrl}" style="width:calc(100% + 80px);height:100%;border:none;margin:0;padding:0;"> </iframe> </div>`);
```
