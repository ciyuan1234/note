```dataviewjs
const page = dv.page(dv.current().file.path);

const incTotal = (page.Inc || []).reduce((sum, item) => sum + (item.IncAmount || 0), 0);
const necTotal = (page.Nec || []).reduce((sum, item) => sum + (item.NecAmount || 0), 0);
const devTotal = (page.Dev || []).reduce((sum, item) => sum + (item.DevAmount || 0), 0);
const entTotal = (page.Ent || []).reduce((sum, item) => sum + (item.EntAmount || 0), 0);

const balance = incTotal - necTotal - devTotal - entTotal;

const file = app.vault.getAbstractFileByPath(page.file.path);
if (file) {
    app.fileManager.processFrontMatter(file, (frontmatter) => {
        Object.assign(frontmatter, {
            IncTotalCount: incTotal,
            NecTotalCount: necTotal,
            DevTotalCount: devTotal,
            EntTotalCount: entTotal,
            Balance: balance
        });
    });
}

dv.table(
    ["收入来源合计", "必需支出合计", "发展支出合计", "休闲支出合计", "当日结余"],
    [[
        `📥 ¥${incTotal.toLocaleString()}`,
        `🏠 -¥${necTotal.toLocaleString()}`,
        `🏃 -¥${devTotal.toLocaleString()}`,
        `🎵 -¥${entTotal.toLocaleString()}`,
        `💰 ${balance >= 0 ? '' : ''}¥${balance.toLocaleString()}`
    ]]
);
```
