```dataviewjs
const endDate = new Date();
const startDate = new Date();
startDate.setDate(startDate.getDate() - 180);

const heatmapData = [];
for (let i = 0; i < 180; i++) {
    const date = new Date(startDate);
    date.setDate(date.getDate() + i);
    heatmapData.push({
        date: date.toISOString().split('T')[0],
        value: 0
    });
}

dv.pages().forEach(page => {
    const createdDate = new Date(page.file.ctime);
    // 只统计在时间范围内的数据
    if (createdDate >= startDate && createdDate <= endDate) {
        const dateStr = createdDate.toISOString().split('T')[0];
        const dayData = heatmapData.find(d => d.date === dateStr);
        if (dayData) {
            dayData.value += 1;
        }
    }
});

const baseUrl = 'https://ytmaps.vip/heatmap-view-main.html'; 
const data = encodeURIComponent(JSON.stringify(heatmapData));
const widgetUrl = `${baseUrl}?data=${data}`;

dv.paragraph(`<div style="width:100%;height:200px;overflow:hidden;">
    <iframe src="${widgetUrl}" style="width:100%;height:100%;border:none;">
    </iframe>
</div>`);
```
