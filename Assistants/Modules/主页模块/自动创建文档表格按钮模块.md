```js-engine
/**
 * Initial Setup 
 **/
// Summoning Meta Bind
const mb = engine.getPlugin('obsidian-meta-bind-plugin').api;

/**
 * Function to create buttons
 **/
function createButton(lb, t, i, fn, fdp) {
    const buttonConfig = {
        label: lb,
        icon: 'activity', // Lucide.dev 'activity' icon's name
        style: 'primary',
        tooltip: t,
        id: i,
        hidden: true,
        actions: [
            { type: 'inlineJS', code: `!app.vault.getFolderByPath('${fdp}') ? await app.vault.createFolder('${fdp}') : '';` },
            { type: 'createNote', fileName: fn, folderPath: fdp, openNote: false }
        ]
    };
    const buttonOptions = {
        declaration: buttonConfig,
        position: undefined,
        isPreview: false
    };
    const button = mb.createButtonMountable(context.file.path, buttonOptions);
    return mb.wrapInMDRC(button, container, component);
};

/**
 * JS Engine Table Setup 
 **/
// Create an array of rows
const rows = [...Array(4).keys()];

// Headers of the table
const headers = [
    "<center><h6>Row Number</h6></center>",
    "<center><h6>Folder Path</h6></center>",
    "<center><h6>File Name</h6></center>",
    "<center><h6>Inline Button</h6></center>"
];

let buttons = [];
let body = [];

// Create rows and buttons
rows.forEach((row, i) => {
    const rowNumber = (i + 1).toString().padStart(2, '0');
    const fileName = `领域 ${rowNumber}`;
    const folderPath = `Documents/I.P.A.R.A/${fileName}`;
    const buttonLabel = `Create ${fileName}`;
    const buttonTooltip = `... in ${folderPath}`;
    const buttonId = `MBCreateNote${rowNumber}`;
    const inlineButton = "`" + `BUTTON[${buttonId}]` + "`";

    buttons.push(createButton(buttonLabel, buttonTooltip, buttonId, fileName, folderPath));
    body.push([
        `<center>${rowNumber}</center>`,
        `<center>${folderPath}</center>`, 
        `<center>${fileName}</center>`, 
        `<center>${inlineButton}</center>`
    ]);
});

/**
 * JS Engine Markdown Builder 
 **/
// Create and return the markdown table
const mdBuilder = engine.markdown.createBuilder();
mdBuilder.createTable(headers, body);
return mdBuilder;
```
