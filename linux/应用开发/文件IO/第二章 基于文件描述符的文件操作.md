# 一、文件IO

## 1.关于Linux文件io
- 在Linux系统下，用于对文件操作的库函数叫做文件I/O
- 主要包括open（）/close（）/read（）/write（）/lseek（）相应的的系统调用（准确来说是对系统调用的封装的库函数）
## 2.文件描述符
- 文件描述符，也可以叫文件标记符，用来标记文件的底层设备
- 文件描述符是一个**非负整数** ，当打开一个已存在的文件或者创建一个新文件时，内核向进程返回一个文件描述符
- 每个程序运行后，操作系统都会默认打开三个文件，**标准输入、标注输出、标准错误输出** ，文件描述符分别为**0，1，2** 
- 标准输入对应的设备一般为**键盘** 
- 标准输入与标准错误输出设备一般为**显示器** 
```cpp
#include<stdio.h>
#include<sys/types.h>
#include<sys/stat.h>
#include<fcntl.h>
#include<errno.h>
#include<unistd.h>

int main(int argc,char*argv[]){
write(1,"helloworld\n",11); //通过标准输出项屏幕打印helloworld
return 0;
}
```
# 二、Linux文件操作
## 1.open函数
### 1.1open函数说明
**函数功能：** 打开文件，并得到文件描述符
**函数原型：** 
- int open(const char\*pathname,int flags)
- int open(const char\*pathname,int flags,mode_t mode)
- ![[Pasted image 20260406221240.png]]
- 头文件说明：
```cpp
 #include<sys/types.h>
 #include<sys/stat.h>
 #include<fcntl.h>
```
- 返回值
  - **成功：** 返回文件描述符
  - **失败：** 返回-1，并将错误码保存到errno
```cpp
// 包含标准输入输出头文件，用于 printf、fprintf、perror 函数
#include<stdio.h>
// 包含系统数据类型头文件，open 函数需要的类型定义
#include<sys/types.h>
// 包含文件状态相关头文件，open 函数权限相关宏定义
#include<sys/stat.h>
// 包含文件控制头文件，open 函数的声明所在
#include<fcntl.h>
// main函数必须写 int argc, char *argv[] 才能接收命令行参数！
int main(int argc, char *argv[])
{
    // 判断命令行参数是否正确
    // 程序名是第1个参数，文件路径是第2个参数，所以必须等于2
    if(argc != 2)
    {
        // 参数错误，打印使用方法到标准错误流
        fprintf(stderr, "Usage: %s <pathname>\n", argv[0]);
        return -1; // 异常退出
    }
    // 定义文件描述符变量
    int fd;
    // 调用 open 系统调用
    // argv[1]：命令行传入的文件路径
    // O_RDONLY：以只读方式打开文件
    // 返回值：成功返回文件描述符（非负整数），失败返回 -1
    fd = open(argv[1], O_RDONLY);
    // 判断 open 是否执行失败
    if(fd == -1)
    {
        // 打印系统级错误信息（会自动提示文件不存在、权限不足等）
        perror("open()");
        return -1; // 异常退出
    }
    // 打开成功，打印文件描述符的值
    printf("fd=%d\n", fd);
    // 关闭文件（好习惯，避免文件描述符泄漏）
    close(fd);
    return 0; // 正常退出
}
```

## 一、read函数
```cpp
//函数头文件
#include<unistd.h>
//函数原型
size_t read(int fd,void*buf,size_t count);
//函数功能：从文件中读取数据保存缓冲区中
```
**示例代码：**
```cpp
// 标准输入输出头文件：提供 printf、fprintf、perror 等函数
#include <stdio.h>
// Unix 标准头文件：提供 read、write、close、usleep 等系统调用
#include <unistd.h>
// 系统数据类型头文件：定义 pid_t、ssize_t、size_t 等系统专用类型
#include <sys/types.h>
// 文件控制头文件：提供 open 函数、O_RDONLY 等文件打开标志
#include <fcntl.h>
// 标准库头文件：提供通用工具函数（这里用于程序规范）
#include <stdlib.h>
/**
 * 主函数
 * argc：命令行参数个数（包含程序自身）
 * argv：命令行参数数组（字符串指针数组）
 */
int main(int argc, char *argv[]) {
    int fd;                 // 文件描述符：操作系统用于标识打开的文件
    ssize_t rbytes = 0;     // 存储 read() 实际读取的字节数
                            // ssize_t：有符号整型，读取失败时返回-1
    char buffer[64] = {0};  // 数据缓冲区：用于存放从文件读取的内容
                            // 初始化为全0，保证字符串结尾有'\0'，打印安全
    // -------------------------- 步骤1：检查命令行参数 --------------------------
    // 判断用户是否传入了文件名参数
    // 正确用法：./程序名 文件名 → 一共2个参数，所以argc必须等于2
    if (argc != 2) {
        // 向标准错误输出流打印使用提示
        fprintf(stderr, "Usage: %s <pathname>\n", argv[0]);
        return -1;  // 参数错误，程序退出，返回-1表示异常
    }
    // -------------------------- 步骤2：打开文件 --------------------------
    // open()：系统调用，打开指定文件
    // 参数1：要打开的文件路径（来自命令行参数）
    // 参数2：O_RDONLY 表示以【只读方式】打开文件
    // 返回值：成功返回 >=0 的文件描述符，失败返回 -1
    fd = open(argv[1], O_RDONLY);
    if (fd == -1) {
        perror("open() failed");  // 打印打开文件失败的具体原因（如文件不存在）
        return -1;
    }
    // -------------------------- 步骤3：读取文件内容 --------------------------
    // read()：系统调用，从打开的文件中读取数据
    // 参数1：文件描述符（指定从哪个文件读）
    // 参数2：存放读取数据的缓冲区
    // 参数3：最多读取的字节数（这里设置为10字节）
    // 返回值：成功返回实际读取的字节数，失败返回-1
    rbytes = read(fd, buffer, 10);
    if (rbytes == -1) {
        perror("read() failed");  // 打印读取失败原因
        close(fd);               // 读取失败也要关闭已打开的文件，避免资源泄漏
        return -1;
    }
    // -------------------------- 步骤4：打印读取结果 --------------------------
    printf("成功读取 %zd 字节数据\n", rbytes);  // %zd：专门用于打印 ssize_t 类型
    printf("读取到的内容：%s\n", buffer);      // 输出缓冲区内容
    // -------------------------- 步骤5：关闭文件 --------------------------
    // close()：关闭文件描述符，释放系统资源
    // 打开的文件必须关闭，否则会造成文件描述符泄漏
    close(fd);
    return 0;  // 程序正常退出
}
```
